# R supplemental package manifest for fair-data-machine
# One package name per line.
# Blank lines and lines starting with # are ignored.
# Uncomment any package line to include it in the image build.

# Core (recommended defaults)
tidyverse   # Core data wrangling + visualization meta-package
data.table  # High-performance data manipulation for large tables
arrow       # Parquet/Feather interoperability and fast columnar IO
DBI         # Standard database interface used by many backends
duckdb      # DuckDB database connector for SQL analytics in R
RPostgres   # Native PostgreSQL client for DB read/write workflows
jsonlite    # JSON parsing/serialization for APIs and metadata
yaml        # YAML read/write for configs and workflow settings
readxl      # Import Excel files (.xls/.xlsx) without Java
haven       # Read SPSS/Stata/SAS files for legacy datasets
sdcMicro    # Statistical disclosure control and anonymization tools

# Extended (uncomment if/when needed)
sf          # Simple Features support for vector geospatial data
terra       # Raster/vector geospatial analysis and processing
targets     # Reproducible pipeline orchestration for R projects
renv        # Project-local package environment and lockfile manager

# Visualization (uncomment as needed)
plotly      # Interactive charts from ggplot2 or native plotly API
leaflet     # Interactive web maps with tile layers and markers
patchwork   # Compose multiple ggplot charts into one layout
gt          # Publication-quality formatted tables
ggthemes    # Additional ggplot themes and styling utilities
scales      # Axis/legend formatting helpers for ggplot2
highcharter # Highcharts-based interactive visualizations

# LLM integration
ellmer      # Unified interface for OpenAI/Anthropic/Google/Ollama-style chat workflows
rollama     # Native Ollama integration for local model inference from R
tidyllm     # Tidyverse-friendly multi-provider LLM interaction patterns
openai      # Direct OpenAI (ChatGPT) API client for provider-specific control
httr2       # Modern HTTP client for custom REST calls to any LLM endpoint
